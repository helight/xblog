SetScrollBar
 

  原型：extern void SetDispBar(int sum, int cur);
            
  用法：#include <system.h>
  
  功能：显示滚动条
  
  说明：调用后在屏幕左侧图标区显示百分比滚动条。
        sum为总长度，cur为当前位置。
  
  举例：

      // SetScrollBar.c
      
      #include <system.h>

      main()
      {
        int i;
        
        clrscr();
        printf("Scroll Bar Test");
        for(i=0;i<60;i+=5)
        {
          SetScrollBar(60,i);
          sleep(1);
        }
        
        getchar();
        return 0;
      }
