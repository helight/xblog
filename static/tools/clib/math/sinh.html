sinh
 

  原型：extern float sinh(float x);
  
  用法：#include <math.h>
  
  功能：计算x（弧度表示）的双曲正弦值。
  
  说明：sinh(x)=(e^x-e^(-x))/2。
  
  举例：


      // sinh.c
      
      #include <syslib.h>
      #include <math.h>

      main()
      {
        float x;
        
        clrscr();        // clear screen
        textmode(0x00);  // 6 lines per LCD screen
        
        x=PI/4.;
        printf("sinh(%.4f)=%.4f\n",x,sinh(x));
        
        getchar();
        return 0;
      }
